cmake_minimum_required(VERSION 2.8)

FIND_PACKAGE(SWIG REQUIRED)
INCLUDE(${SWIG_USE_FILE})
SET(CMAKE_SWIG_OUTDIR "${CMAKE_CURRENT_SOURCE_DIR}/generated/org/openh264")

INCLUDE_DIRECTORIES("${OPENH264_HOME}/codec/api/svc")
LINK_DIRECTORIES("${OPENH264_HOME}")
INCLUDE_DIRECTORIES(${CMAKE_CURRENT_SOURCE_DIR})
SET_SOURCE_FILES_PROPERTIES(codec_api.i PROPERTIES CPLUSPLUS ON)
SET_PROPERTY(SOURCE codec_api.i PROPERTY SWIG_FLAGS "-package" "org.openh264")
SWIG_ADD_MODULE(codec_api java codec_api.i)
SWIG_LINK_LIBRARIES(codec_api openh264)
